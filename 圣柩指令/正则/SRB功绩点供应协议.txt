    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const dataInput = document.getElementById('data-input');
            const itemListContainer = document.getElementById('item-list-container');
            const detailWrapper = document.getElementById('item-detail-wrapper');
            const detailPlaceholder = document.getElementById('detail-placeholder');
            const detailError = document.getElementById('detail-error');
            const detailContentArea = document.getElementById('detail-content-area');
            const categoryBarContainer = document.getElementById('category-bar-container');
            const cartItemsContainer = document.getElementById('cart-items-container');
            const cartTotalPrice = document.getElementById('cart-total-price');
            const checkoutBtn = document.getElementById('checkout-btn');
            const meritPointsDisplay = document.getElementById('merit-points-display');
            const commanderRatingDisplay = document.getElementById('commander-rating-display');

            let cart = [];
            let allItems = [];
            let currentTotalCost = 0;

            const ratingLevels = {
                '新晋指挥官': 1,
                '认证指挥官': 2,
                '资深指挥官': 3,
                '王牌指挥官': 4,
                '先驱': 5,
            };

            function mapSortToCategoryKey(sortString) {
                const mapping = {
                    '战略物资': 'strategic',
                    '技术蓝图与许可': 'tech',
                    '人事调配': 'personnel',
                    '心智慰藉品': 'consumable'
                };
                for (const key in mapping) {
                    if (sortString.includes(key)) return mapping[key];
                }
                return 'other';
            }

            function initializeProtocol(rawData) {
                allItems = parseSupplyItems(rawData);
                if (allItems.length > 0) {
                    displayCategoryButtons(allItems);
                    displayItemsList(allItems);
                    setupCategoryFilters();
                    clearDetailView();
                    renderCart();
                } else {
                    showError();
                }
            }

            function parseSupplyItems(data) {
                let contentToParse = data;
                const mspContentMatch = data.match(/<MSP>([\s\S]+?)<\/MSP>/);

                if (mspContentMatch) {
                    contentToParse = mspContentMatch[1];
                }

                const parsedItems = [];
                const itemRegex = /<item>\s*<supply>([\s\S]*?)<\/supply>\s*<sort>([\s\S]*?)<\/sort>\s*<price>([\s\S]*?)<\/price>(?:\s*<unit>([\s\S]*?)<\/unit>)?\s*<info>([\s\S]*?)<\/info>\s*<rank>([\s\S]*?)<\/rank>\s*<\/item>/g;
                let match;

                while ((match = itemRegex.exec(contentToParse)) !== null) {
                    const [, supply, sort, price, unit, info, rank] = match.map(s => s ? s.trim() : s);

                    const originalSort = sort || '其他';
                    const categoryKey = mapSortToCategoryKey(originalSort);
                    const priceValue = parseFloat(price.replace(/[^0-9.-]+/g,"")) || 0;

                    parsedItems.push({
                        id: `item-${parsedItems.length}`,
                        supply: supply,
                        sort: categoryKey,
                        originalSort: originalSort,
                        price: price,
                        priceValue: priceValue,
                        unit: unit || null,
                        info: info || '无详细信息。',
                        rank: rank || '无'
                    });
                }
                return parsedItems;
            }

            function displayCategoryButtons(items) {
                const categories = new Map();
                categories.set('all', '全部');

                items.forEach(item => {
                    if (!Array.from(categories.keys()).includes(item.sort)) {
                        categories.set(item.sort, item.originalSort);
                    }
                });

                categoryBarContainer.innerHTML = '';
                categories.forEach((value, key) => {
                    const button = document.createElement('button');
                    button.className = 'category-btn';
                    button.dataset.category = key;
                    button.textContent = value;
                    if (key === 'all') {
                        button.classList.add('active');
                    }
                    categoryBarContainer.appendChild(button);
                });
            }

            function displayItemsList(items) {
                itemListContainer.innerHTML = '';
                items.forEach(item => {
                    const card = document.createElement('div');
                    card.className = 'item-card';
                    card.id = item.id;
                    card.dataset.sort = item.sort;
                    card.innerHTML = `<div class="item-name">${item.supply}</div><div class="item-price">${item.price}</div>`;
                    card.addEventListener('click', () => {
                        displayItemDetail(item);
                        document.querySelectorAll('.item-card').forEach(c => c.classList.remove('active'));
                        card.classList.add('active');
                    });
                    itemListContainer.appendChild(card);
                });
            }

            function displayItemDetail(item) {
                detailPlaceholder.style.display = 'none';
                detailError.style.display = 'none';
                detailContentArea.style.display = 'flex';

                let unitHtml = '';
                if (item.unit) {
                    unitHtml = `
                    <div class="detail-section">
                        <div class="detail-label">单位</div>
                        <div class="detail-value">${item.unit}</div>
                    </div>`;
                }

                detailContentArea.innerHTML = `
                    <div class="detail-header">
                        <div class="detail-name">${item.supply}</div>
                        <div class="detail-price">${item.price}</div>
                    </div>
                    <div class="detail-section">
                        <div class="detail-label">分类</div>
                        <div class="detail-value">${item.originalSort}</div>
                    </div>
                    ${unitHtml}
                    <div class="detail-section">
                        <div class="detail-label">信息</div>
                        <div class="detail-value">${item.info}</div>
                    </div>
                    <div class="detail-section">
                        <div class="detail-label">兑换要求</div>
                        <div class="detail-value rank-value">${item.rank}</div>
                    </div>
                    <div class="detail-section">
                        <div class="detail-label">数量</div>
                        <div class="quantity-selector">
                            <button class="quantity-btn" id="decrease-quantity">-</button>
                            <input type="number" class="quantity-input" id="quantity-input" value="1" min="1">
                            <button class="quantity-btn" id="increase-quantity">+</button>
                        </div>
                    </div>
                    <button class="add-to-cart-btn" id="add-to-cart">加入选择</button>`;

                document.getElementById('decrease-quantity').addEventListener('click', () => {
                    const input = document.getElementById('quantity-input');
                    let value = parseInt(input.value);
                    if (value > 1) input.value = value - 1;
                });

                document.getElementById('increase-quantity').addEventListener('click', () => {
                    const input = document.getElementById('quantity-input');
                    input.value = parseInt(input.value) + 1;
                });

                document.getElementById('add-to-cart').addEventListener('click', () => {
                    const quantity = parseInt(document.getElementById('quantity-input').value);
                    addToCart(item, quantity);
                });
            }

            function addToCart(item, quantity) {
                const userRatingText = commanderRatingDisplay.textContent.trim().replace(/['"\[\]]+/g, '');
                const userRatingLevel = ratingLevels[userRatingText] || 0;
                const itemRatingText = item.rank.trim();
                const itemRatingLevel = ratingLevels[itemRatingText] || 0;

                if (userRatingLevel < itemRatingLevel) {
                    alert('指挥官评级不足!');
                    return;
                }

                const existingItem = cart.find(cartItem => cartItem.id === item.id);
                if (existingItem) {
                    existingItem.quantity += quantity;
                } else {
                    cart.push({ ...item, quantity: quantity });
                }
                renderCart();
            }

            function renderCart() {
                cartItemsContainer.innerHTML = '';
                let total = 0;
                cart.forEach(item => {
                    const cartItemElement = document.createElement('div');
                    cartItemElement.className = 'cart-item';
                    cartItemElement.innerHTML = `
                        <span class="cart-item-name">${item.supply}</span>
                        <div class="cart-item-quantity">
                            <button class="quantity-btn cart-quantity-btn" data-id="${item.id}" data-change="-1">-</button>
                            <span>${item.quantity}</span>
                            <button class="quantity-btn cart-quantity-btn" data-id="${item.id}" data-change="1">+</button>
                        </div>
                        <span class="cart-item-price">${(item.priceValue * item.quantity).toFixed(2)}</span>
                    `;
                    cartItemsContainer.appendChild(cartItemElement);
                    total += item.priceValue * item.quantity;
                });
                currentTotalCost = total;
                cartTotalPrice.textContent = total.toFixed(2);

                document.querySelectorAll('.cart-quantity-btn').forEach(button => {
                    button.addEventListener('click', (e) => {
                        const itemId = e.target.dataset.id;
                        const change = parseInt(e.target.dataset.change);
                        updateCartQuantity(itemId, change);
                    });
                });
            }

            function updateCartQuantity(itemId, change) {
                const itemIndex = cart.findIndex(item => item.id === itemId);
                if (itemIndex > -1) {
                    cart[itemIndex].quantity += change;
                    if (cart[itemIndex].quantity <= 0) {
                        cart.splice(itemIndex, 1);
                    }
                }
                renderCart();
            }

            function setupCategoryFilters() {
                const categoryButtons = document.querySelectorAll('.category-btn');
                categoryButtons.forEach(button => {
                    button.addEventListener('click', () => {
                        const category = button.dataset.category;
                        categoryButtons.forEach(btn => btn.classList.remove('active'));
                        button.classList.add('active');
                        const allItemCards = document.querySelectorAll('.item-card');
                        allItemCards.forEach(card => {
                            if (category === 'all' || card.dataset.sort === category) {
                                card.classList.remove('hidden');
                            } else {
                                card.classList.add('hidden');
                            }
                        });
                        clearDetailView();
                    });
                });
            }

            function clearDetailView(){
                detailContentArea.style.display = 'none';
                detailError.style.display = 'none';
                detailPlaceholder.style.display = 'block';
                detailPlaceholder.innerHTML = '<p>← 从列表中选择项目查看详情</p>';
                document.querySelectorAll('.item-card').forEach(c => c.classList.remove('active'));
            }

            function showError() {
                detailContentArea.style.display = 'none';
                detailPlaceholder.style.display = 'none';
                detailError.style.display = 'block';
            }

            function decodeHtml(html) {
                var txt = document.createElement("textarea");
                txt.innerHTML = html;
                return txt.value;
            }

            checkoutBtn.addEventListener('click', () => {
                if (cart.length === 0) {
                    alert('选择为空!');
                    return;
                }

                const currentMeritPoints = parseFloat(meritPointsDisplay.textContent) || 0;
                if (currentMeritPoints < currentTotalCost) {
                    alert('功绩点不足!');
                    return;
                }

                const itemsParts = cart.map(item => {
                    let unitString = item.unit;
                    if (!unitString) {
                        const priceParts = item.price.split('/');
                        unitString = priceParts.length > 1 ? priceParts[1] : '份';
                    }

                    const unitMatch = unitString.match(/^(\d*\.?\d+)(.*)$/);
                    if (unitMatch) {
                        const unitValue = parseFloat(unitMatch[1]);
                        const unitName = unitMatch[2] || '';
                        const totalAmount = item.quantity * unitValue;
                        return `${item.supply}，${totalAmount}${unitName}`;
                    } else {
                        return `${item.supply}，${item.quantity}${unitString}`;
                    }
                });

                const checkoutMessage = `花费${currentTotalCost.toFixed(2)}MP，购买清单：${itemsParts.join('；')}。`;

                if (typeof triggerSlash !== 'undefined') {
                    triggerSlash(`/send ${checkoutMessage}`);
                    cart = [];
                    renderCart();
                } else {
                    console.log('SillyTavern API (triggerSlash) not found. Data to send:');
                    console.log(checkoutMessage);
                    alert('未在SillyTavern环境中运行。请在控制台查看生成的数据。');
                }
            });

            const interval = 50, timeout = 2000; let elapsedTime = 0;
            const dataObserver = setInterval(() => {
                const rawData = dataInput.value;
                elapsedTime += interval;
                if (rawData && rawData.trim() !== '$' + '1') {
                    clearInterval(dataObserver);
                    const decodedData = decodeHtml(rawData);
                    initializeProtocol(decodedData);
                } else if (elapsedTime >= timeout) {
                    clearInterval(dataObserver);
                    showError();
                }
            }, interval);
        });
    </script>
</body>
